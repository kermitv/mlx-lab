    find "$HOME/Library/LaunchAgents" -maxdepth 1 -type f 2>/dev/null | sed "s#^$HOME#~#" || true
    echo
    echo "# grep likely relevant launchagent contents"
    for f in "$HOME"/Library/LaunchAgents/*.plist; do
      [ -f "$f" ] || continue
      if grep -Eiq "mlx|python|uvicorn|fastapi|openclaw|qwen" "$f" 2>/dev/null; then
        echo
        echo "## $f"
        plutil -p "$f" 2>/dev/null || cat "$f"
      fi
    done
    echo
    echo "# launchctl user services hint"
    launchctl list 2>/dev/null | egrep -i "mlx|python|openclaw|qwen|uvicorn|fastapi" || true
  }
